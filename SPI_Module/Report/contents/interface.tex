\subsubsection*{Buf\_P2S}
\textbf{Parameter:} \texttt{N = 8} (default).\\
\textbf{Inputs:}
\begin{itemize}
	\item \texttt{Input[N-1:0]}: parallel input data (N bits).
	\item \texttt{load}: signal used to load data into the buffer.
	\item \texttt{clk}: clock signal for data shifting.
	\item \texttt{ena}: shift enable signal.
\end{itemize}
\textbf{Outputs:}
\begin{itemize}
	\item \texttt{Output}: serial output data (1 bit).
	\item \texttt{Buffer[N-1:0]}: buffer content in parallel form.
\end{itemize}

\begin{figure}[H]
	\centering
	\includegraphics[width=0.6\textwidth]{graphics/Tai_2.pdf}
	\caption{\texttt{Buf\_S2P}}
	\label{fig:buf_s2p}
\end{figure}

\subsubsection*{Buf\_S2P}
\textbf{Parameter:} \texttt{N = 8} (default).\\
\textbf{Inputs:}
\begin{itemize}
	\item \texttt{Input}: serial input data (1 bit).
	\item \texttt{clk}: clock signal.
	\item \texttt{ena}: shift enable signal.
\end{itemize}
\textbf{Outputs:}
\begin{itemize}
	\item \texttt{Output[N-1:0]}: parallel data after receiving all N bits.
\end{itemize}

\subsubsection*{crc5}
\textbf{Inputs:}
\begin{itemize}
	\item \texttt{clk}: clock signal.
	\item \texttt{reset}: resets the CRC value to 0.
	\item \texttt{enable}: enables CRC update.
	\item \texttt{serial\_in}: serial input data (1 bit).
\end{itemize}
\textbf{Outputs:}
\begin{itemize}
	\item \texttt{crc\_out[4:0]}: current CRC-5 value.
\end{itemize}

\begin{figure}[H]
	\centering
	\includegraphics[width=0.6\textwidth]{graphics/Tai_4.pdf}
	\caption{\texttt{crc5\_checker}}
	\label{fig:crc5_checker}
\end{figure}

\subsubsection*{crc5\_checker}
\textbf{Inputs:}
\begin{itemize}
	\item \texttt{clk}: clock signal.
	\item \texttt{reset}: resets the checker logic.
	\item \texttt{enable}: enables CRC checking on the received serial stream.
	\item \texttt{serial\_in}: serial input data (1 bit).
\end{itemize}
\textbf{Outputs:}
\begin{itemize}
	\item \texttt{data\_valid}: indicates that the received data is valid when the CRC is correct.
	\item \texttt{check\_crc\_out[4:0]}: current checked CRC value, if observation is required.
\end{itemize}

\subsection{Slave Module}

The main block is \texttt{SPI\_Slave\_CRC}, which internally uses the submodules \texttt{Parallel\_to\_serial}, \texttt{serial\_to\_parallel}, \texttt{crc5}, and \texttt{crc5\_checker}.
\begin{figure}[H]
	\centering
	\includegraphics[width=0.85\textwidth]{graphics/Tai_6.pdf}
	\caption{Slave top-level block diagram}
	\label{fig:slave_top}
\end{figure}

\subsubsection*{SPI\_Slave\_CRC}
\textbf{Parameters:} \texttt{n=8}, \texttt{crc\_len=5}, \texttt{total\_len=n+crc\_len}, \texttt{cnum=4}, \texttt{CPOL=0}, \texttt{CPHA=0}.\\
\textbf{Inputs:}
\begin{itemize}
	\item \texttt{n\_rst}: active-low reset.
	\item \texttt{data\_in[n-1:0]}: parallel data provided by the Slave for MISO transmission.
	\item \texttt{load}: loads data into the P2S block.
	\item \texttt{sclk}: SPI clock from the Master.
	\item \texttt{MOSI}: serial data from the Master.
	\item \texttt{n\_cs}: active-low chip select.
\end{itemize}
\textbf{Outputs:}
\begin{itemize}
	\item \texttt{data\_out[n-1:0]}: received parallel data after collecting all N bits.
	\item \texttt{MISO}: serial data returned to the Master, including payload and CRC bits.
	\item \texttt{rx\_data\_valid}: indicates that the received data is valid when the CRC is correct.
	\item \texttt{rx\_done}: indicates that the full frame, including data and CRC, has been received.
	\item \texttt{crc\_out[crc\_len-1:0]}: CRC generated for the transmitted data.
	\item \texttt{ready}: indicates that the Slave is ready for transmission or reception.
\end{itemize}

\subsubsection*{Parallel\_to\_serial}
\textbf{Function:} converts the parallel input \texttt{data\_in[n-1:0]} into the serial stream \texttt{data\_out} when \texttt{load} and \texttt{ena} are asserted.

\subsubsection*{serial\_to\_parallel}
\textbf{Function:} captures serial input data \texttt{data\_in} and reconstructs it into parallel output \texttt{data\_out[n-1:0]} when \texttt{ena} is active.
