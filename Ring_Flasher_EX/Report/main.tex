\subsubsection{Flowchart Example}
\begin{figure}[h]
\centering
\begin{tikzpicture}[
    node distance=1.5cm,
    startstop/.style={rectangle, rounded corners, minimum width=3cm, minimum height=1cm, text centered, draw=black, fill=red!30},
    process/.style={rectangle, minimum width=3cm, minimum height=1cm, text centered, draw=black, fill=orange!30},
    decision/.style={diamond, minimum width=3cm, minimum height=1cm, text centered, draw=black, fill=green!30},
    arrow/.style={thick,->,>=stealth}
]

    % Nodes
    \node (start) [startstop] {Start};
    \node (init) [process, below of=start] {Initialize GPIO};
    \node (loop) [process, below of=init] {Start Animation Loop};
    \node (led1) [process, below of=loop] {Turn ON LED1};
    \node (delay1) [process, below of=led1] {Delay 500ms};
    \node (led2) [process, below of=delay1] {Turn OFF LED1\\Turn ON LED2};
    \node (delay2) [process, below of=led2] {Delay 500ms};
    \node (check) [decision, below of=delay2] {Continue?};
    \node (stop) [startstop, left of=check, xshift=-3cm] {Stop};
    
    % Arrows
    \draw [arrow] (start) -- (init);
    \draw [arrow] (init) -- (loop);
    \draw [arrow] (loop) -- (led1);
    \draw [arrow] (led1) -- (delay1);
    \draw [arrow] (delay1) -- (led2);
    \draw [arrow] (led2) -- (delay2);
    \draw [arrow] (delay2) -- (check);
    \draw [arrow] (check) -- node[anchor=east] {No} (stop);
    \draw [arrow] (check.east) -- ++(2,0) |- node[anchor=west, near start] {Yes} (loop.east);

\end{tikzpicture}
\caption{LED Animation Program Flowchart}
\end{figure}
